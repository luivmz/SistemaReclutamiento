### 1.1 Agregar Tabla: RESPUESTAS_POSTULANTES

```sql
CREATE TABLE respuestas_postulantes (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    postulante_id BIGINT NOT NULL,
    pregunta_id BIGINT NOT NULL,
    respuesta_texto VARCHAR(500),
    es_correcta BOOLEAN,
```

### 2.2 Módulo de Reportes

Crear vistas SQL para KPIs:
```sql
-- KPI: Tiempo promedio por etapa
CREATE VIEW kpi_tiempo_etapas AS
SELECT 
    oferta_id,
    estado,
    AVG(DATEDIFF(day, fecha_postulacion, CURRENT_TIMESTAMP)) as dias_promedio
FROM postulantes
GROUP BY oferta_id, estado;

-- KPI: Tasa de aprobación por oferta
CREATE VIEW kpi_tasa_aprobacion AS
SELECT 
    o.titulo,
    COUNT(p.id) as total_postulantes,
    COUNT(CASE WHEN p.aprobado = TRUE THEN 1 END) as aprobados,
    ROUND(COUNT(CASE WHEN p.aprobado = TRUE THEN 1 END) * 100.0 / COUNT(p.id), 2) as porcentaje_aprobacion
FROM ofertas_laborales o
LEFT JOIN postulantes p ON o.id = p.oferta_id
GROUP BY o.id, o.titulo;
```

### 4.1 Índices Base de Datos

```sql

CREATE INDEX idx_postulantes_oferta_estado ON postulantes(oferta_id, estado);
```
